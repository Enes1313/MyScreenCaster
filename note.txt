APK_PATH=app/build/outputs/apk/debug/app-debug.apk
ZIPALIGNED_APK=app/build/outputs/apk/debug/app-debug-aligned.apk

# zipalign (SDK build-tools içinde)
zipalign -v -p 4 $APK_PATH $ZIPALIGNED_APK

# apksigner (SDK build-tools içinde)
apksigner sign --ks ~/.android/debug.keystore --ks-pass pass:android --out app-debug-signed.apk $ZIPALIGNED_APK



adb install -r app-debug-signed.apk
